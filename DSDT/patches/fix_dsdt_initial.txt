into method label _DSM parent_label SAT0 replace_content
begin
If (PCIC (Arg0))\n
{\n
Return (PCID (Arg0, Arg1, Arg2, Arg3))\n
}\n
If (CondRefOf (\_SB.PCI0.SAT0.SDSM))\n
{\n
Return (SDSM)\n
}\n
Return (Zero)\n
end;

into method label ADBG replace_content
begin
If (CondRefOf (MDBG))\n
{\n
Return (MDBG)\n
}\n
Return (Zero)\n
end;
